Attribute VB_Name = "modCommentFormatting"
Option Explicit

'------------------------------------------------------------------------------
' Module : modCommentFormatting
' Purpose : Builds structured chart and dashboard comment text.
'------------------------------------------------------------------------------

Private Const COMMENT_BULLET As String = "• "
Private Const SECTION_EXTENSIONS As String = "EXTENSIONS TO RAISE"
Private Const SECTION_SCHEDULED As String = "SCHEDULED THIS WEEK"
Private Const SECTION_DUE As String = "DUE"
Private Const MAX_BULLET_LINE_LEN As Long = 120


'------------------------------------------------------------------------------
' Purpose : Formats a decimal proportion as a percentage to two decimal places.
'------------------------------------------------------------------------------
Public Function FormatOccurrencePercent(ByVal proportion As Double) As String

    FormatOccurrencePercent = Format$(proportion * 100, "0.00") & "%"

End Function


'------------------------------------------------------------------------------
' Purpose : Returns imported extension percent unless the task was reforecasted.
'------------------------------------------------------------------------------
Public Function GetEffectiveExistingExtensionPercent(ByRef occurrence As TaskOccurrence) As Double

    If occurrence.wasAutoPushed Or occurrence.wasAutoPulled Or occurrence.GroupedDirection <> 0 Then
        GetEffectiveExistingExtensionPercent = 0
    Else
        GetEffectiveExistingExtensionPercent = occurrence.existingExtensionPercent
    End If

End Function


'------------------------------------------------------------------------------
' Purpose : Returns imported extension amount unless the task was reforecasted.
'------------------------------------------------------------------------------
Public Function GetEffectiveExistingExtensionAmount(ByRef occurrence As TaskOccurrence) As Double

    If occurrence.wasAutoPushed Or occurrence.wasAutoPulled Or occurrence.GroupedDirection <> 0 Then
        GetEffectiveExistingExtensionAmount = 0
    Else
        GetEffectiveExistingExtensionAmount = occurrence.existingExtensionAmount
    End If

End Function


'------------------------------------------------------------------------------
' Purpose : Formats an extension or interval amount for comment display.
'------------------------------------------------------------------------------
Public Function FormatExtensionAmount(ByRef occurrence As TaskOccurrence, _
                                      ByVal amount As Double) As String

    Dim amountText As String

    If occurrence.intervalType = "DD" Then
        If amount = Int(amount) Then
            amountText = CStr(CLng(amount))
        Else
            amountText = Format$(amount, "0.##")
        End If

        FormatExtensionAmount = amountText & " DD"
    Else
        FormatExtensionAmount = FormatHoursToHMM(amount) & " " & occurrence.intervalType
    End If

End Function


'------------------------------------------------------------------------------
' Purpose : Builds the task code and description label for a comment bullet.
'------------------------------------------------------------------------------
Public Function BuildOccurrenceTaskLabel(ByRef occurrence As TaskOccurrence) As String

    BuildOccurrenceTaskLabel = BuildOccurrenceTaskLabelWithMaxDesc(occurrence, MAX_DESC_LEN)

End Function


'------------------------------------------------------------------------------
' Purpose : Builds a task label with the description truncated to a max length.
'------------------------------------------------------------------------------
Private Function BuildOccurrenceTaskLabelWithMaxDesc(ByRef occurrence As TaskOccurrence, _
                                                     ByVal maxDescLen As Long) As String

    If occurrence.isHighlighted Then
        BuildOccurrenceTaskLabelWithMaxDesc = occurrence.taskCode & " — " & _
                                              TruncateDesc(occurrence.taskDescription, maxDescLen) & _
                                              " [Critical]"
    Else
        BuildOccurrenceTaskLabelWithMaxDesc = occurrence.taskCode & " — " & _
                                              TruncateDesc(occurrence.taskDescription, maxDescLen)
    End If

End Function


'------------------------------------------------------------------------------
' Purpose : Builds one comment bullet truncated to a single display line.
'------------------------------------------------------------------------------
Private Function BuildSingleLineBullet(ByRef occurrence As TaskOccurrence, _
                                       ByVal tailText As String) As String

    Dim criticalTag As String
    Dim fixedLength As Long
    Dim maxDescLen As Long

    If occurrence.isHighlighted Then
        criticalTag = " [Critical]"
    Else
        criticalTag = vbNullString
    End If

    fixedLength = Len(COMMENT_BULLET) + _
                  Len(occurrence.taskCode) + _
                  Len(" — ") + _
                  Len(criticalTag) + _
                  Len(" | ") + _
                  Len(tailText)

    maxDescLen = MAX_BULLET_LINE_LEN - fixedLength

    If maxDescLen < 8 Then maxDescLen = 8

    BuildSingleLineBullet = COMMENT_BULLET & _
                            BuildOccurrenceTaskLabelWithMaxDesc(occurrence, maxDescLen) & _
                            " | " & tailText

End Function


'------------------------------------------------------------------------------
' Purpose : Builds the days or hours remaining text for one occurrence.
'------------------------------------------------------------------------------
Public Function BuildOccurrenceTimeRemainingText(ByRef occurrence As TaskOccurrence) As String

    If occurrence.intervalType = "DD" Then

        Dim daysRemaining As Long
        daysRemaining = CLng(occurrence.firstDueDate) - CLng(Date)

        If daysRemaining < 0 Then
            If Abs(daysRemaining) = 1 Then
                BuildOccurrenceTimeRemainingText = "1 day overdue"
            Else
                BuildOccurrenceTimeRemainingText = CStr(Abs(daysRemaining)) & " days overdue"
            End If
        ElseIf daysRemaining = 0 Then
            BuildOccurrenceTimeRemainingText = "due today"
        ElseIf daysRemaining = 1 Then
            BuildOccurrenceTimeRemainingText = "1 day remaining"
        Else
            BuildOccurrenceTimeRemainingText = CStr(daysRemaining) & " days remaining"
        End If

    Else

        If occurrence.lifeRemaining < 0 Then
            BuildOccurrenceTimeRemainingText = _
                FormatHoursToHMM(Abs(occurrence.lifeRemaining)) & " hrs overdue"
        Else
            BuildOccurrenceTimeRemainingText = _
                FormatHoursToHMM(occurrence.lifeRemaining) & " hrs remaining"
        End If

    End If

    If occurrence.intervalType = "DD" Then
        BuildOccurrenceTimeRemainingText = _
            BuildOccurrenceTimeRemainingText & " | " & BuildOccurrenceDueText(occurrence)
    End If

End Function


'------------------------------------------------------------------------------
' Purpose : Returns total extension percent used for grouping extension comments.
'------------------------------------------------------------------------------
Public Function GetOccurrenceTotalExtensionPercent(ByRef occurrence As TaskOccurrence) As Double

    GetOccurrenceTotalExtensionPercent = _
        GetEffectiveExistingExtensionPercent(occurrence) + occurrence.extensionPercentUsed

End Function


'------------------------------------------------------------------------------
' Purpose : Builds a grouping key from total extension for one occurrence.
'------------------------------------------------------------------------------
Public Function BuildOccurrenceTotalExtensionGroupKey(ByRef occurrence As TaskOccurrence) As String

    BuildOccurrenceTotalExtensionGroupKey = _
        FormatOccurrencePercent(GetOccurrenceTotalExtensionPercent(occurrence))

End Function


'------------------------------------------------------------------------------
' Purpose : Adds one extension occurrence into a grouped week dictionary.
'------------------------------------------------------------------------------
Public Sub AddExtensionOccurrenceToWeekGroups(ByVal weekGroups As Object, _
                                              ByRef occurrence As TaskOccurrence)

    Dim weekKey As String
    Dim intervalLabel As String
    Dim groupKey As String

    weekKey = BuildExtensionDueWeekKey(occurrence.originalWeek)
    intervalLabel = BuildOccurrenceIntervalLabel(occurrence)
    groupKey = intervalLabel & "|" & BuildOccurrenceTotalExtensionGroupKey(occurrence)

    If Not weekGroups.Exists(weekKey) Then
        weekGroups.Add weekKey, CreateObject("Scripting.Dictionary")
    End If

    Dim weekGroupDictionary As Object
    Set weekGroupDictionary = weekGroups(weekKey)

    If Not weekGroupDictionary.Exists(groupKey) Then
        Dim groupEntry As Object
        Set groupEntry = CreateObject("Scripting.Dictionary")

        groupEntry.Add "IntervalLabel", intervalLabel
        groupEntry.Add "TaskCodes", occurrence.taskCode
        groupEntry.Add "SummaryText", BuildExtensionRaiseSummaryText(occurrence)
        groupEntry.Add "TaskCount", 1

        weekGroupDictionary.Add groupKey, groupEntry
    Else
        Dim existingEntry As Object
        Set existingEntry = weekGroupDictionary(groupKey)

        existingEntry("TaskCodes") = existingEntry("TaskCodes") & ", " & occurrence.taskCode
        existingEntry("TaskCount") = CLng(existingEntry("TaskCount")) + 1
    End If

End Sub


'------------------------------------------------------------------------------
' Purpose : Builds one grouped extension bullet line for chart/dashboard comments.
'------------------------------------------------------------------------------
Public Function BuildExtensionGroupBulletLine(ByVal weekKey As String, _
                                              ByVal groupEntry As Object) As String

    BuildExtensionGroupBulletLine = COMMENT_BULLET & CStr(groupEntry("TaskCodes")) & _
                                    " | Raise by " & weekKey & _
                                    " | " & CStr(groupEntry("SummaryText"))

End Function


'------------------------------------------------------------------------------
' Purpose : Builds grouped extension comment text for one due week.
'------------------------------------------------------------------------------
Public Function BuildExtensionWeekGroupedComment(ByVal weekKey As String, _
                                                 ByVal extensionCount As Long, _
                                                 ByVal weekGroupDictionary As Object) As String

    If extensionCount <= 0 Then
        BuildExtensionWeekGroupedComment = vbNullString
        Exit Function
    End If

    Dim result As String
    Dim intervalLabels As Object
    Dim sortedLabels() As String
    Dim labelIndex As Long
    Dim groupKey As Variant

    result = "Raise by " & weekKey & " (" & CStr(extensionCount) & ")"

    Set intervalLabels = CreateObject("Scripting.Dictionary")

    For Each groupKey In weekGroupDictionary.Keys
        Dim groupEntry As Object
        Set groupEntry = weekGroupDictionary(groupKey)

        Dim intervalLabel As String
        intervalLabel = CStr(groupEntry("IntervalLabel"))

        If intervalLabels.Exists(intervalLabel) Then
            intervalLabels(intervalLabel) = _
                CLng(intervalLabels(intervalLabel)) + CLng(groupEntry("TaskCount"))
        Else
            intervalLabels.Add intervalLabel, CLng(groupEntry("TaskCount"))
        End If
    Next groupKey

    sortedLabels = SortIntervalLabels(intervalLabels)

    For labelIndex = LBound(sortedLabels) To UBound(sortedLabels)

        result = result & vbLf & vbLf & sortedLabels(labelIndex) & _
                 " (" & CStr(intervalLabels(sortedLabels(labelIndex))) & ")"

        For Each groupKey In weekGroupDictionary.Keys
            Set groupEntry = weekGroupDictionary(groupKey)

            If CStr(groupEntry("IntervalLabel")) = sortedLabels(labelIndex) Then
                result = result & vbLf & BuildExtensionGroupBulletLine(weekKey, groupEntry)
            End If
        Next groupKey

    Next labelIndex

    BuildExtensionWeekGroupedComment = result

End Function


'------------------------------------------------------------------------------
' Purpose : Returns interval labels sorted by type and value.
'------------------------------------------------------------------------------
Private Function SortIntervalLabels(ByVal intervalLabels As Object) As String()

    Dim labelArray() As String
    ReDim labelArray(1 To intervalLabels.Count)

    Dim labelKey As Variant
    Dim index As Long

    For Each labelKey In intervalLabels.Keys
        index = index + 1
        labelArray(index) = CStr(labelKey)
    Next labelKey

    Dim outerIndex As Long
    Dim innerIndex As Long

    For outerIndex = 1 To UBound(labelArray) - 1
        For innerIndex = outerIndex + 1 To UBound(labelArray)
            If CompareIntervalLabels(labelArray(innerIndex), labelArray(outerIndex)) < 0 Then
                Dim tempLabel As String
                tempLabel = labelArray(outerIndex)
                labelArray(outerIndex) = labelArray(innerIndex)
                labelArray(innerIndex) = tempLabel
            End If
        Next innerIndex
    Next outerIndex

    SortIntervalLabels = labelArray

End Function


'------------------------------------------------------------------------------
' Purpose : Compares two interval labels for grouped extension sorting.
'------------------------------------------------------------------------------
Private Function CompareIntervalLabels(ByVal leftLabel As String, _
                                       ByVal rightLabel As String) As Long

    Dim leftParts() As String
    Dim rightParts() As String

    leftParts = Split(leftLabel, " ")
    rightParts = Split(rightLabel, " ")

    Dim leftTypeRank As Long
    Dim rightTypeRank As Long

    leftTypeRank = GetIntervalTypeSortRank(leftParts(0))
    rightTypeRank = GetIntervalTypeSortRank(rightParts(0))

    If leftTypeRank <> rightTypeRank Then
        CompareIntervalLabels = leftTypeRank - rightTypeRank
        Exit Function
    End If

    CompareIntervalLabels = Sgn(CDbl(leftParts(1)) - CDbl(rightParts(1)))

End Function


'------------------------------------------------------------------------------
' Purpose : Returns the sort rank for one interval type label.
'------------------------------------------------------------------------------
Private Function GetIntervalTypeSortRank(ByVal intervalType As String) As Long

    Select Case intervalType
        Case "HH": GetIntervalTypeSortRank = 1
        Case "E1": GetIntervalTypeSortRank = 2
        Case "E2": GetIntervalTypeSortRank = 3
        Case "DD": GetIntervalTypeSortRank = 4
        Case Else: GetIntervalTypeSortRank = 5
    End Select

End Function


'------------------------------------------------------------------------------
' Purpose : Builds the movement suffixes for a scheduled task comment.
'------------------------------------------------------------------------------
Public Function BuildOccurrenceScheduledTailText(ByRef occurrence As TaskOccurrence) As String

    Dim tailText As String

    tailText = BuildOccurrenceTimeRemainingText(occurrence)

    If occurrence.GroupedDirection <> 0 Then
        tailText = tailText & " | " & BuildGroupedAlignmentCommentText(occurrence)
    ElseIf occurrence.wasAutoPulled And occurrence.pullPercentUsed > 0 Then
        tailText = tailText & " | Pulled " & FormatOccurrencePullAmount(occurrence) & _
                   " (" & FormatOccurrencePercent(occurrence.pullPercentUsed) & _
                   ") to Maint"
    End If

    BuildOccurrenceScheduledTailText = tailText

End Function


'------------------------------------------------------------------------------
' Purpose : Returns the extension amount added by auto-push.
'------------------------------------------------------------------------------
Public Function GetOccurrenceExtensionAddedAmount(ByRef occurrence As TaskOccurrence) As Double

    GetOccurrenceExtensionAddedAmount = occurrence.extensionPercentUsed * occurrence.intervalValue

End Function


'------------------------------------------------------------------------------
' Purpose : Formats the extension amount added by auto-push.
'------------------------------------------------------------------------------
Public Function FormatOccurrenceExtensionAddedAmount(ByRef occurrence As TaskOccurrence) As String

    FormatOccurrenceExtensionAddedAmount = _
        FormatExtensionAmount(occurrence, GetOccurrenceExtensionAddedAmount(occurrence))

End Function


'------------------------------------------------------------------------------
' Purpose : Builds the raise-row/dashboard extension summary for one occurrence.
'------------------------------------------------------------------------------
Public Function BuildExtensionRaiseSummaryText(ByRef occurrence As TaskOccurrence) As String

    Dim existingPercent As Double
    Dim existingAmount As Double
    Dim totalPercent As Double
    Dim totalAmount As Double

    existingPercent = GetEffectiveExistingExtensionPercent(occurrence)
    existingAmount = GetEffectiveExistingExtensionAmount(occurrence)
    totalPercent = existingPercent + occurrence.extensionPercentUsed
    totalAmount = totalPercent * occurrence.intervalValue

    BuildExtensionRaiseSummaryText = _
        "Extend by " & FormatOccurrenceExtensionAddedAmount(occurrence) & _
        " (" & FormatOccurrencePercent(occurrence.extensionPercentUsed) & ")" & _
        " | Existing Extension " & FormatExtensionAmount(occurrence, existingAmount) & _
        " (" & FormatOccurrencePercent(existingPercent) & ")" & _
        " | Total Extension " & FormatExtensionAmount(occurrence, totalAmount) & _
        " (" & FormatOccurrencePercent(totalPercent) & ")"

End Function

' Purpose : Builds the chart-cell extension summary for one occurrence.

Public Function BuildChartCellExtensionSummaryText(ByRef occurrence As TaskOccurrence) As String

    BuildChartCellExtensionSummaryText = _

End Function

'------------------------------------------------------------------------------

'------------------------------------------------------------------------------
Public Function BuildChartCellScheduledBullet(ByRef occurrence As TaskOccurrence) As String

    Dim tailText As String

    tailText = BuildOccurrenceScheduledTailText(occurrence)

    If occurrence.wasAutoPushed Then
        tailText = "[EXT. REQ. BY] " & Format$(occurrence.originalWeek, "DD/MM/YYYY") & _

    End If

    BuildChartCellScheduledBullet = BuildSingleLineBullet(occurrence, tailText)

End Function


'------------------------------------------------------------------------------
' Purpose : Appends one bullet line to an existing multiline bullet block.
'------------------------------------------------------------------------------
Public Sub AppendCommentBullet(ByRef bulletBlock As String, ByVal bulletLine As String)

    If Len(bulletBlock) > 0 Then
        bulletBlock = bulletBlock & vbLf & bulletLine
    Else
        bulletBlock = bulletLine
    End If

End Sub


'------------------------------------------------------------------------------
' Purpose : Builds one structured comment section.
'------------------------------------------------------------------------------
Public Function BuildCommentSection(ByVal sectionTitle As String, _
                                    ByVal taskCount As Long, _
                                    ByVal bulletBlock As String) As String

    If taskCount <= 0 Then
        BuildCommentSection = vbNullString
        Exit Function
    End If

    BuildCommentSection = sectionTitle & " (" & CStr(taskCount) & ")" & vbLf & bulletBlock

End Function

' Purpose : Builds a structured chart cell comment for scheduled tasks.
'------------------------------------------------------------------------------
Public Function BuildStructuredChartCellComment(ByVal extensionCount As Long, _
                                                 ByVal extensionBullets As String, _
                                                 ByVal scheduledCount As Long, _
                                                 ByVal scheduledBullets As String) As String

    BuildStructuredChartCellComment = _
        BuildCommentSection(SECTION_SCHEDULED, scheduledCount, scheduledBullets)

End Function


'------------------------------------------------------------------------------
' Purpose : Builds the extension-due row comment for one chart week column.
'------------------------------------------------------------------------------
Public Function BuildExtensionDueWeekComment(ByVal extensionCount As Long, _
                                             ByVal weekKey As String, _
                                             ByVal weekGroupDictionary As Object) As String

    BuildExtensionDueWeekComment = _
        BuildExtensionWeekGroupedComment(weekKey, extensionCount, weekGroupDictionary)

End Function


'------------------------------------------------------------------------------
' Purpose : Builds a dashboard interval-group comment with extensions listed first.
'------------------------------------------------------------------------------
Public Function BuildDashboardIntervalGroupComment(ByVal intervalLabel As String, _
                                                   ByVal extensionCount As Long, _
                                                   ByVal extensionBullets As String, _
                                                   ByVal dueCount As Long, _
                                                   ByVal dueBullets As String) As String

    Dim result As String
    Dim extensionSection As String
    Dim dueSection As String

    result = intervalLabel & " (" & CStr(extensionCount + dueCount) & ")"

    extensionSection = BuildCommentSection(SECTION_EXTENSIONS, extensionCount, extensionBullets)
    dueSection = BuildCommentSection(SECTION_DUE, dueCount, dueBullets)

    If Len(extensionSection) > 0 Then
        result = result & vbLf & vbLf & extensionSection
    End If

    If Len(dueSection) > 0 Then
        If Len(extensionSection) = 0 Then
            result = result & vbLf & vbLf
        Else
            result = result & vbLf & vbLf
        End If
        result = result & dueSection
    End If

    BuildDashboardIntervalGroupComment = result

End Function


'------------------------------------------------------------------------------
' Purpose : Builds a dashboard extensions-by-week group comment.
'------------------------------------------------------------------------------
Public Function BuildDashboardExtensionWeekComment(ByVal weekKey As String, _
                                                   ByVal extensionCount As Long, _
                                                   ByVal weekGroupDictionary As Object) As String

    BuildDashboardExtensionWeekComment = _
        BuildExtensionWeekGroupedComment(weekKey, extensionCount, weekGroupDictionary)

End Function
