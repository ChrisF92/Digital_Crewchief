Attribute VB_Name = "modWordExport"
Option Explicit

'------------------------------------------------------------------------------
' Module : modWordExport
' Purpose : Exports selected chart period tasks to a Microsoft Word work pack.
'------------------------------------------------------------------------------

Private Const EXT_TABLE_COLS As Long = 5
Private Const TASK_TABLE_COLS As Long = 8

' Late-bound Word constants.
Private Const WD_COLLAPSE_END As Long = 0
Private Const WD_ORIENT_LANDSCAPE As Long = 1
Private Const WD_AUTO_FIT_WINDOW As Long = 2
Private Const WD_LINE_STYLE_SINGLE As Long = 1
Private Const WD_COLOR_BLACK As Long = 0
Private Const WD_ALIGN_PARAGRAPH_CENTER As Long = 1


'------------------------------------------------------------------------------
' Purpose : Generates a Microsoft Word work pack for the selected aircraft chart period.
' Input : None. Uses ActiveSheet and ActiveCell to determine aircraft and selected period.
' Output : Creates and displays a Word document containing selected-period tasks.
'------------------------------------------------------------------------------
Public Sub GenerateWordWorkPack()

    Dim chartWs As Worksheet
    Set chartWs = ActiveSheet

    If Not IsAircraftChartSheet(chartWs.Name) Then
        MsgBox "Open an aircraft Chart sheet first, select a week/down period, then run this report." & vbCrLf & vbCrLf & _
               "Expected sheet name format: TailNumber" & CHART_SHEET_SUFFIX & vbCrLf & _
               "Current sheet: " & chartWs.Name, _
               vbExclamation, "Generate Work Pack"
        Exit Sub
    End If

    If ActiveCell.Column < CHART_FIRST_WEEK_COL Then
        MsgBox "Select a week/down-period column on the chart first.", _
               vbExclamation, "Generate Work Pack"
        Exit Sub
    End If

    Dim selCol As Long
    selCol = ActiveCell.Column

    Dim firstWeekCol As Long
    Dim lastWeekCol As Long

    firstWeekCol = CHART_FIRST_WEEK_COL
    lastWeekCol = chartWs.Cells(CHART_DATE_ROW, chartWs.Columns.Count).End(xlToLeft).Column

    If selCol < firstWeekCol Or selCol > lastWeekCol Then
        MsgBox "Select a valid week/down-period column.", vbExclamation, "Generate Work Pack"
        Exit Sub
    End If

    ' Determine selected period.
    Dim periodStartCol As Long
    Dim periodEndCol As Long

    periodStartCol = selCol
    periodEndCol = selCol

    If ActiveCell.MergeCells Then
        periodStartCol = ActiveCell.MergeArea.Column
        periodEndCol = ActiveCell.MergeArea.Column + ActiveCell.MergeArea.Columns.Count - 1

    ElseIf IsDownOrMaintHeader(chartWs.Cells(3, selCol).Value) Then

        Do While periodStartCol > firstWeekCol
            If IsDownOrMaintHeader(chartWs.Cells(3, periodStartCol - 1).Value) Then
                periodStartCol = periodStartCol - 1
            Else
                Exit Do
            End If
        Loop

        Do While periodEndCol < lastWeekCol
            If IsDownOrMaintHeader(chartWs.Cells(3, periodEndCol + 1).Value) Then
                periodEndCol = periodEndCol + 1
            Else
                Exit Do
            End If
        Loop

    End If

    If Not IsDate(chartWs.Cells(4, periodStartCol).Value) Or _
       Not IsDate(chartWs.Cells(4, periodEndCol).Value) Then
        MsgBox "The selected period does not contain valid dates on row 4.", _
               vbExclamation, "Generate Work Pack"
        Exit Sub
    End If

    Dim periodStartDate As Date
    Dim periodEndDate As Date

    periodStartDate = CDate(chartWs.Cells(4, periodStartCol).Value)
    periodEndDate = CDate(chartWs.Cells(4, periodEndCol).Value) + 6

    Dim tail As String
    tail = Left$(Trim$(chartWs.Name), Len(Trim$(chartWs.Name)) - Len(Trim$(CHART_SHEET_SUFFIX)))

    Dim periodTitle As String

    If periodStartCol = periodEndCol Then
        periodTitle = CStr(chartWs.Cells(3, periodStartCol).Value) & " - " & _
                      Format$(chartWs.Cells(4, periodStartCol).Value, "DD/MM/YYYY")
    Else
        periodTitle = "Down / Maintenance Period - " & _
                      Format$(periodStartDate, "DD/MM/YYYY") & _
                      " to " & Format$(periodEndDate, "DD/MM/YYYY")
    End If

    On Error GoTo ErrHandler

    BeginFastMode

    Dim taskOccurrences() As TaskOccurrence
    Dim totalOccurrences As Long

    If Not TryBuildGroupedTaskOccurrencesForTail(tail, taskOccurrences, totalOccurrences) Then
        MsgBox "Could not rebuild task data for " & tail & "." & vbCrLf & vbCrLf & _
               "Check that the aircraft is on the Rotation sheet and has an import loaded.", _
               vbExclamation, "Generate Work Pack"
        GoTo CleanExit
    End If

    Dim extensionIndexes() As Long
    Dim scheduledIndexes() As Long
    Dim extensionCount As Long
    Dim scheduledCount As Long

    CollectWordExportOccurrences _
        taskOccurrences, _
        totalOccurrences, _
        periodStartDate, _
        periodEndDate, _
        extensionIndexes, _
        extensionCount, _
        scheduledIndexes, _
        scheduledCount

    If extensionCount = 0 And scheduledCount = 0 Then
        MsgBox "No tasks or extensions found for the selected week/down period.", _
               vbInformation, "Generate Work Pack"
        GoTo CleanExit
    End If

    ' Create Word document using late binding.
    Dim wdApp As Object
    Dim wdDoc As Object
    Dim wdRange As Object
    Dim wordWasRunning As Boolean

    wordWasRunning = False

    On Error Resume Next
    Set wdApp = GetObject(, "Word.Application")
    If Not wdApp Is Nothing Then wordWasRunning = True
    On Error GoTo ErrHandler

    If wdApp Is Nothing Then Set wdApp = CreateObject("Word.Application")

    If wdApp Is Nothing Then
        MsgBox "Could not start Microsoft Word.", vbExclamation, "Generate Work Pack"
        GoTo CleanExit
    End If

    wdApp.Visible = True

    Set wdDoc = wdApp.Documents.Add
    Set wdRange = wdDoc.Range

    With wdDoc.PageSetup
        .Orientation = WD_ORIENT_LANDSCAPE
        .LeftMargin = 36
        .RightMargin = 36
        .TopMargin = 54
        .BottomMargin = 54
    End With

    wdRange.Text = tail & " Maintenance Work Pack" & vbCr & _
                   periodTitle & vbCr & _
                   "Generated: " & Format$(Now, "DD/MM/YYYY HH:NN") & vbCr & _
                   "Scheduled tasks: " & scheduledCount & vbCr & _
                   "Extensions required: " & extensionCount & vbCr

    With wdDoc.Paragraphs(1).Range.Font
        .Name = "Arial"
        .Bold = True
        .Size = 18
    End With

    With wdDoc.Paragraphs(2).Range.Font
        .Name = "Arial"
        .Bold = True
        .Size = 12
    End With

    Dim paraIndex As Long
    For paraIndex = 3 To 5
        With wdDoc.Paragraphs(paraIndex).Range.Font
            .Name = "Arial"
            .Size = 10
        End With
    Next paraIndex

    Set wdRange = wdDoc.Content
    wdRange.Collapse WD_COLLAPSE_END

    AppendWordSectionHeading wdDoc, "EXTENSIONS REQUIRED"
    Set wdRange = wdDoc.Content
    wdRange.Collapse WD_COLLAPSE_END

    If extensionCount = 0 Then
        wdRange.InsertAfter "No extensions required for this selected period." & vbCr
    Else
        WriteExtensionsWordTable _
            wdDoc, _
            wdRange, _
            taskOccurrences, _
            extensionIndexes, _
            extensionCount
    End If

    Set wdRange = wdDoc.Content
    wdRange.Collapse WD_COLLAPSE_END
    wdRange.InsertParagraphAfter
    wdRange.Collapse WD_COLLAPSE_END

    AppendWordSectionHeading wdDoc, "SCHEDULED TASKS"
    Set wdRange = wdDoc.Content
    wdRange.Collapse WD_COLLAPSE_END

    If scheduledCount = 0 Then
        wdRange.InsertAfter "No scheduled tasks for this selected period." & vbCr
    Else
        WriteScheduledTasksWordTable _
            wdDoc, _
            wdRange, _
            taskOccurrences, _
            scheduledIndexes, _
            scheduledCount
    End If

    MsgBox "Word work pack created for " & tail & "." & Chr$(10) & _
           scheduledCount & " scheduled task(s), " & extensionCount & " extension(s).", _
           vbInformation, "Generate Work Pack"

CleanExit:
    EndFastMode
    Exit Sub

ErrHandler:
    On Error Resume Next
    If Not wdDoc Is Nothing Then wdDoc.Close SaveChanges:=False
    If Not wordWasRunning And Not wdApp Is Nothing Then wdApp.Quit
    On Error GoTo 0

    EndFastMode

    MsgBox "Word export failed." & vbCrLf & vbCrLf & _
           "Error " & Err.Number & ": " & Err.Description, _
           vbExclamation, "Generate Work Pack"

End Sub


'------------------------------------------------------------------------------
' Purpose : Collects extension and scheduled occurrence indexes for a selected period.
' Notes : Extensions are keyed by originalWeek (raise-by week). Scheduled tasks are
' keyed by scheduledWeek (including push destination weeks).
'------------------------------------------------------------------------------
Private Sub CollectWordExportOccurrences(ByRef taskOccurrences() As TaskOccurrence, _
                                         ByVal totalOccurrences As Long, _
                                         ByVal periodStartDate As Date, _
                                         ByVal periodEndDate As Date, _
                                         ByRef extensionIndexes() As Long, _
                                         ByRef extensionCount As Long, _
                                         ByRef scheduledIndexes() As Long, _
                                         ByRef scheduledCount As Long)

    Dim occurrenceIndex As Long
    Dim seenScheduled As Object

    Set seenScheduled = CreateObject("Scripting.Dictionary")

    extensionCount = 0
    scheduledCount = 0

    If totalOccurrences > 0 Then
        ReDim extensionIndexes(1 To totalOccurrences)
        ReDim scheduledIndexes(1 To totalOccurrences)
    End If

    For occurrenceIndex = 1 To totalOccurrences

        Dim occurrence As TaskOccurrence
        occurrence = taskOccurrences(occurrenceIndex)

        If occurrence.isDeferredFault Then GoTo NextOccurrence

        If IsExtensionRequiredInPeriod(occurrence, periodStartDate, periodEndDate) Then
            extensionCount = extensionCount + 1
            extensionIndexes(extensionCount) = occurrenceIndex
        End If

        If IsScheduledInPeriod(occurrence, periodStartDate, periodEndDate) Then

            Dim dedupeKey As String
            dedupeKey = GetTaskOccurrenceKey(occurrence)

            If Not seenScheduled.Exists(dedupeKey) Then
                seenScheduled.Add dedupeKey, True
                scheduledCount = scheduledCount + 1
                scheduledIndexes(scheduledCount) = occurrenceIndex
            End If

        End If

NextOccurrence:
    Next occurrenceIndex

    If extensionCount > 1 Then
        SortOccurrenceIndexesByDueDate taskOccurrences, extensionIndexes, extensionCount
    End If

    If scheduledCount > 1 Then
        SortOccurrenceIndexesByDueDate taskOccurrences, scheduledIndexes, scheduledCount
    End If

End Sub


'------------------------------------------------------------------------------
' Purpose : True when an auto-pushed task requires an extension in the selected period.
'------------------------------------------------------------------------------
Private Function IsExtensionRequiredInPeriod(ByRef occurrence As TaskOccurrence, _
                                             ByVal periodStartDate As Date, _
                                             ByVal periodEndDate As Date) As Boolean

    If Not occurrence.wasAutoPushed Then
        IsExtensionRequiredInPeriod = False
        Exit Function
    End If

    If CLng(occurrence.originalWeek) = CLng(occurrence.scheduledWeek) Then
        IsExtensionRequiredInPeriod = False
        Exit Function
    End If

    IsExtensionRequiredInPeriod = IsWeekStartInPeriod( _
        occurrence.originalWeek, _
        periodStartDate, _
        periodEndDate)

End Function


'------------------------------------------------------------------------------
' Purpose : True when a task is scheduled into the selected period.
'------------------------------------------------------------------------------
Private Function IsScheduledInPeriod(ByRef occurrence As TaskOccurrence, _
                                     ByVal periodStartDate As Date, _
                                     ByVal periodEndDate As Date) As Boolean

    IsScheduledInPeriod = IsWeekStartInPeriod( _
        occurrence.scheduledWeek, _
        periodStartDate, _
        periodEndDate)

End Function


'------------------------------------------------------------------------------
' Purpose : True when a Monday week-start falls inside the selected period dates.
'------------------------------------------------------------------------------
Private Function IsWeekStartInPeriod(ByVal weekStart As Date, _
                                     ByVal periodStartDate As Date, _
                                     ByVal periodEndDate As Date) As Boolean

    IsWeekStartInPeriod = (weekStart >= periodStartDate And weekStart <= periodEndDate)

End Function


'------------------------------------------------------------------------------
' Purpose : Sorts occurrence indexes by first due date, then task code.
'------------------------------------------------------------------------------
Private Sub SortOccurrenceIndexesByDueDate(ByRef taskOccurrences() As TaskOccurrence, _
                                           ByRef indexes() As Long, _
                                           ByVal itemCount As Long)

    Dim i As Long
    Dim j As Long
    Dim tempIndex As Long

    For i = 1 To itemCount - 1
        For j = i + 1 To itemCount
            If CompareOccurrencesForWordExport( _
                taskOccurrences(indexes(j)), _
                taskOccurrences(indexes(i))) < 0 Then

                tempIndex = indexes(i)
                indexes(i) = indexes(j)
                indexes(j) = tempIndex
            End If
        Next j
    Next i

End Sub


'------------------------------------------------------------------------------
' Purpose : Comparison helper for Word export row ordering.
'------------------------------------------------------------------------------
Private Function CompareOccurrencesForWordExport(ByRef leftOccurrence As TaskOccurrence, _
                                                 ByRef rightOccurrence As TaskOccurrence) As Long

    If leftOccurrence.firstDueDate < rightOccurrence.firstDueDate Then
        CompareOccurrencesForWordExport = -1
    ElseIf leftOccurrence.firstDueDate > rightOccurrence.firstDueDate Then
        CompareOccurrencesForWordExport = 1
    Else
        CompareOccurrencesForWordExport = StrComp( _
            UCase$(Trim$(leftOccurrence.taskCode)), _
            UCase$(Trim$(rightOccurrence.taskCode)), _
            vbTextCompare)
    End If

End Function


'------------------------------------------------------------------------------
' Purpose : Writes the extensions-required Word table.
'------------------------------------------------------------------------------
Private Sub WriteExtensionsWordTable(ByVal wdDoc As Object, _
                                     ByVal wdRange As Object, _
                                     ByRef taskOccurrences() As TaskOccurrence, _
                                     ByRef extensionIndexes() As Long, _
                                     ByVal extensionCount As Long)

    Dim tbl As Object
    Dim rowIndex As Long
    Dim occurrence As TaskOccurrence

    Set tbl = wdDoc.Tables.Add(wdRange, extensionCount + 1, EXT_TABLE_COLS)

    SetWordTableCellText tbl, 1, 1, "Task Code"
    SetWordTableCellText tbl, 1, 2, "When Due"
    SetWordTableCellText tbl, 1, 3, "Current Extension"
    SetWordTableCellText tbl, 1, 4, "Required Extension"
    SetWordTableCellText tbl, 1, 5, "Total Extension"

    For rowIndex = 1 To extensionCount
        occurrence = taskOccurrences(extensionIndexes(rowIndex))

        SetWordTableCellText tbl, rowIndex + 1, 1, occurrence.taskCode
        SetWordTableCellText tbl, rowIndex + 1, 2, Format$(occurrence.firstDueDate, "DD/MM/YYYY")
        SetWordTableCellText tbl, rowIndex + 1, 3, FormatExtensionAmountAndPercent( _
            occurrence, _
            occurrence.existingExtensionAmount, _
            occurrence.existingExtensionPercent)
        SetWordTableCellText tbl, rowIndex + 1, 4, FormatExtensionAmountAndPercent( _
            occurrence, _
            GetOccurrenceExtensionAddedAmount(occurrence), _
            occurrence.extensionPercentUsed)
        SetWordTableCellText tbl, rowIndex + 1, 5, FormatExtensionAmountAndPercent( _
            occurrence, _
            occurrence.existingExtensionAmount + GetOccurrenceExtensionAddedAmount(occurrence), _
            occurrence.existingExtensionPercent + occurrence.extensionPercentUsed)
    Next rowIndex

    StyleWordExportTable tbl

End Sub


'------------------------------------------------------------------------------
' Purpose : Writes the scheduled-tasks Word table.
'------------------------------------------------------------------------------
Private Sub WriteScheduledTasksWordTable(ByVal wdDoc As Object, _
                                         ByVal wdRange As Object, _
                                         ByRef taskOccurrences() As TaskOccurrence, _
                                         ByRef scheduledIndexes() As Long, _
                                         ByVal scheduledCount As Long)

    Dim tbl As Object
    Dim rowIndex As Long
    Dim occurrence As TaskOccurrence

    Set tbl = wdDoc.Tables.Add(wdRange, scheduledCount + 1, TASK_TABLE_COLS)

    SetWordTableCellText tbl, 1, 1, "When Due"
    SetWordTableCellText tbl, 1, 2, "Interval Type"
    SetWordTableCellText tbl, 1, 3, "Task Code"
    SetWordTableCellText tbl, 1, 4, "Task Sequence"
    SetWordTableCellText tbl, 1, 5, "Actual Interval"
    SetWordTableCellText tbl, 1, 6, "Description"
    SetWordTableCellText tbl, 1, 7, "Extension"
    SetWordTableCellText tbl, 1, 8, "Serial Number"

    For rowIndex = 1 To scheduledCount
        occurrence = taskOccurrences(scheduledIndexes(rowIndex))

        SetWordTableCellText tbl, rowIndex + 1, 1, Format$(occurrence.firstDueDate, "DD/MM/YYYY")
        SetWordTableCellText tbl, rowIndex + 1, 2, occurrence.intervalType
        SetWordTableCellText tbl, rowIndex + 1, 3, occurrence.taskCode
        SetWordTableCellText tbl, rowIndex + 1, 4, occurrence.sequenceNumber
        SetWordTableCellText tbl, rowIndex + 1, 5, FormatActualInterval(occurrence)
        SetWordTableCellText tbl, rowIndex + 1, 6, occurrence.taskDescription
        SetWordTableCellText tbl, rowIndex + 1, 7, FormatScheduledExtensionCell(occurrence)
        SetWordTableCellText tbl, rowIndex + 1, 8, occurrence.serialNumber
    Next rowIndex

    StyleWordExportTable tbl

End Sub


'------------------------------------------------------------------------------
' Purpose : Formats amount and percentage for Word extension columns.
'------------------------------------------------------------------------------
Private Function FormatExtensionAmountAndPercent(ByRef occurrence As TaskOccurrence, _
                                                 ByVal amount As Double, _
                                                 ByVal proportion As Double) As String

    FormatExtensionAmountAndPercent = FormatExtensionAmount(occurrence, amount) & _
                                      " (" & FormatOccurrencePercent(proportion) & ")"

End Function


'------------------------------------------------------------------------------
' Purpose : Formats the actual interval value for the scheduled-tasks table.
'------------------------------------------------------------------------------
Private Function FormatActualInterval(ByRef occurrence As TaskOccurrence) As String

    FormatActualInterval = FormatExtensionAmount(occurrence, occurrence.intervalValue)

End Function


'------------------------------------------------------------------------------
' Purpose : Formats the extension column for a scheduled task row.
'------------------------------------------------------------------------------
Private Function FormatScheduledExtensionCell(ByRef occurrence As TaskOccurrence) As String

    If occurrence.wasAutoPushed Then
        FormatScheduledExtensionCell = FormatExtensionAmountAndPercent( _
            occurrence, _
            GetOccurrenceExtensionAddedAmount(occurrence), _
            occurrence.extensionPercentUsed)
    Else
        FormatScheduledExtensionCell = vbNullString
    End If

End Function


'------------------------------------------------------------------------------
' Purpose : Writes one Word table cell value without leaving a trailing cell mark.
'------------------------------------------------------------------------------
Private Sub SetWordTableCellText(ByVal tbl As Object, _
                                 ByVal rowIndex As Long, _
                                 ByVal columnIndex As Long, _
                                 ByVal cellText As String)

    Dim cellRange As Object
    Set cellRange = tbl.Cell(rowIndex, columnIndex).Range

    ' Exclude the end-of-cell marker, then replace the visible cell text.
    cellRange.MoveEnd 1, -1   ' wdCharacter = 1
    cellRange.Text = cellText

End Sub


'------------------------------------------------------------------------------
' Purpose : Applies shared formatting to a Word export table.
'------------------------------------------------------------------------------
Private Sub StyleWordExportTable(ByVal tbl As Object)

    Dim borderIndex As Long

    On Error Resume Next
    tbl.AutoFitBehavior WD_AUTO_FIT_WINDOW

    For borderIndex = 1 To 6
        With tbl.Borders(borderIndex)
            .LineStyle = WD_LINE_STYLE_SINGLE
            .LineWidth = 4
            .Color = WD_COLOR_BLACK
        End With
    Next borderIndex
    On Error GoTo 0

    With tbl.Range.Font
        .Name = "Arial"
        .Size = 8
        .Bold = False
    End With

    With tbl.Rows(1).Range.Font
        .Bold = True
        .Size = 9
    End With

    tbl.Rows(1).Range.ParagraphFormat.Alignment = WD_ALIGN_PARAGRAPH_CENTER

    On Error Resume Next
    tbl.Rows(1).Shading.BackgroundPatternColor = RGB(220, 220, 220)
    On Error GoTo 0

End Sub


'------------------------------------------------------------------------------
' Purpose : Appends a bold section heading at the end of the Word document.
'------------------------------------------------------------------------------
Private Sub AppendWordSectionHeading(ByVal wdDoc As Object, ByVal headingText As String)

    Dim wdRange As Object
    Dim headingStart As Long

    Set wdRange = wdDoc.Content
    wdRange.Collapse WD_COLLAPSE_END

    If wdRange.Start > 0 Then
        wdRange.InsertAfter vbCr
        wdRange.Collapse WD_COLLAPSE_END
    End If

    headingStart = wdRange.Start
    wdRange.InsertAfter headingText & vbCr

    With wdDoc.Range(headingStart, headingStart + Len(headingText))
        .Font.Name = "Arial"
        .Font.Bold = True
        .Font.Size = 12
    End With

End Sub


'------------------------------------------------------------------------------
' Purpose : Determines whether a chart header value represents a down or maintenance week.
' Input : headerValue - value from the chart header row, usually row 3.
' Output : True if the value is "DOWN" or starts with "MAINT WK"; otherwise false.
'------------------------------------------------------------------------------
Private Function IsDownOrMaintHeader(ByVal headerValue As Variant) As Boolean

    Dim s As String
    s = UCase$(Trim$(CStr(headerValue)))

    IsDownOrMaintHeader = (s = "DOWN" Or Left$(s, 8) = "MAINT WK")

End Function


'------------------------------------------------------------------------------
' Purpose : Determines whether a worksheet name follows the aircraft chart
' naming convention.
' Input : sheetName - worksheet name to test.
' Output : True if the sheet name ends with CHART_SHEET_SUFFIX;
' otherwise False.
'------------------------------------------------------------------------------
Private Function IsAircraftChartSheet(ByVal sheetName As String) As Boolean

    Dim cleanSheetName As String
    Dim cleanSuffix As String

    cleanSheetName = UCase$(Trim$(sheetName))
    cleanSuffix = UCase$(Trim$(CHART_SHEET_SUFFIX))

    If Len(cleanSuffix) = 0 Then
        IsAircraftChartSheet = False
    Else
        IsAircraftChartSheet = (Right$(cleanSheetName, Len(cleanSuffix)) = cleanSuffix)
    End If

End Function
