Attribute VB_Name = "modTaskOccurrenceFormatting"
Option Explicit

'------------------------------------------------------------------------------
' Module : modTaskOccurrenceFormatting
' Purpose : Formats TaskOccurrence records for dashboard and chart comments.
'------------------------------------------------------------------------------

'------------------------------------------------------------------------------
' Purpose : Builds a short interval label such as "HH 25", "E1 50", or "DD 180".
'------------------------------------------------------------------------------
Public Function BuildOccurrenceIntervalLabel(ByRef occurrence As TaskOccurrence) As String

    If occurrence.intervalValue = Int(occurrence.intervalValue) Then
        BuildOccurrenceIntervalLabel = occurrence.intervalType & " " & _
                                       CStr(CLng(occurrence.intervalValue))
    Else
        BuildOccurrenceIntervalLabel = occurrence.intervalType & " " & _
                                       Format$(occurrence.intervalValue, "0.##")
    End If

End Function


'------------------------------------------------------------------------------
' Purpose : Builds the due/life remaining text for a task occurrence.
'------------------------------------------------------------------------------
Public Function BuildOccurrenceDueText(ByRef occurrence As TaskOccurrence) As String

    If occurrence.intervalType = "DD" Then
        BuildOccurrenceDueText = "Due: " & Format$(occurrence.firstDueDate, "DD/MM/YYYY")
    Else
        BuildOccurrenceDueText = "Due: 0:00 " & occurrence.intervalType
    End If

End Function


'------------------------------------------------------------------------------
' Purpose : Formats grouped alignment distance for pull-to-align comments.
'------------------------------------------------------------------------------
Public Function FormatGroupedAlignmentPullAmount(ByRef occurrence As TaskOccurrence) As String

    If occurrence.intervalType = "DD" Then

        Dim alignmentDays As Long
        alignmentDays = GetOccurrenceGroupedAlignmentDays(occurrence)

        If alignmentDays = 1 Then
            FormatGroupedAlignmentPullAmount = "1 day"
        Else
            FormatGroupedAlignmentPullAmount = CStr(alignmentDays) & " days"
        End If

    Else
        FormatGroupedAlignmentPullAmount = _
            FormatHoursToHMM(occurrence.groupedAlignmentAmount) & " " & occurrence.intervalType
    End If

End Function


'------------------------------------------------------------------------------
' Purpose : Builds the grouped-alignment suffix for scheduled task comments.
'------------------------------------------------------------------------------
Public Function BuildGroupedAlignmentCommentText(ByRef occurrence As TaskOccurrence) As String

    If occurrence.GroupedDirection = 0 Then
        BuildGroupedAlignmentCommentText = vbNullString
        Exit Function
    End If

    Dim directionText As String

    If occurrence.GroupedDirection < 0 Then
        directionText = "Pulled by "
    Else
        directionText = "Pushed by "
    End If

    BuildGroupedAlignmentCommentText = directionText & _
                                       FormatGroupedAlignmentPullAmount(occurrence) & _
                                       " to align with " & occurrence.groupedWithTaskCode

End Function


'------------------------------------------------------------------------------
' Purpose : Returns the absolute amount pulled forward for a maintenance pull.
'------------------------------------------------------------------------------
Public Function GetOccurrencePullAmount(ByRef occurrence As TaskOccurrence) As Double

    GetOccurrencePullAmount = occurrence.pullPercentUsed * occurrence.intervalValue

End Function


'------------------------------------------------------------------------------
' Purpose : Formats a maintenance pull amount for display in comments.
'------------------------------------------------------------------------------
Public Function FormatOccurrencePullAmount(ByRef occurrence As TaskOccurrence) As String

    Dim pullAmount As Double
    pullAmount = GetOccurrencePullAmount(occurrence)

    If occurrence.intervalType = "DD" Then
        If pullAmount = Int(pullAmount) Then
            FormatOccurrencePullAmount = CStr(CLng(pullAmount)) & " days"
        Else
            FormatOccurrencePullAmount = Format$(pullAmount, "0.##") & " days"
        End If
    Else
        FormatOccurrencePullAmount = FormatHoursToHMM(pullAmount) & " hrs"
    End If

End Function


'------------------------------------------------------------------------------
' Purpose : Returns the alignment distance between grouped original and scheduled weeks.
'------------------------------------------------------------------------------
Public Function GetOccurrenceGroupedAlignmentDays(ByRef occurrence As TaskOccurrence) As Long

    GetOccurrenceGroupedAlignmentDays = Abs(CLng(occurrence.originalWeek) - CLng(occurrence.scheduledWeek))

End Function


'------------------------------------------------------------------------------
' Purpose : Formats grouped alignment movement for chart comments.
'------------------------------------------------------------------------------
Public Function FormatOccurrenceGroupedAlignmentAmount(ByRef occurrence As TaskOccurrence) As String

    If occurrence.intervalType = "DD" Then

        Dim alignmentDays As Long
        alignmentDays = GetOccurrenceGroupedAlignmentDays(occurrence)

        If alignmentDays = 1 Then
            FormatOccurrenceGroupedAlignmentAmount = "1 day"
        Else
            FormatOccurrenceGroupedAlignmentAmount = CStr(alignmentDays) & " days"
        End If

    Else

        If occurrence.groupedAlignmentAmount > 0 Then
            FormatOccurrenceGroupedAlignmentAmount = _
                FormatHoursToHMM(occurrence.groupedAlignmentAmount) & " hrs"
        Else
            FormatOccurrenceGroupedAlignmentAmount = "0:00 hrs"
        End If

    End If

End Function


'------------------------------------------------------------------------------
' Purpose : Returns the dashboard/chart summary key for an extension due week.
'------------------------------------------------------------------------------
Public Function BuildExtensionDueWeekKey(ByVal originalWeek As Date) As String

    BuildExtensionDueWeekKey = Format$(originalWeek, "DD/MM/YYYY")

End Function


'------------------------------------------------------------------------------
' Purpose : Builds a dashboard detail bullet for one occurrence.
'------------------------------------------------------------------------------
Public Function BuildDashboardOccurrenceDetail(ByRef occurrence As TaskOccurrence) As String

    BuildDashboardOccurrenceDetail = BuildOccurrenceScheduledBullet(occurrence)

End Function


'------------------------------------------------------------------------------
' Purpose : Builds one extension-due bullet for chart and dashboard summaries.
'------------------------------------------------------------------------------
Public Function BuildExtensionDueDetailLine(ByRef occurrence As TaskOccurrence) As String

    BuildExtensionDueDetailLine = BuildExtensionRaiseBullet(occurrence)

End Function


'------------------------------------------------------------------------------
' Purpose : Adds one occurrence to a dashboard summary bucket if it has not
' already been added.
'------------------------------------------------------------------------------
Public Sub AddOccurrenceToDashboardBucket(ByVal summaryDictionary As Object, _
                                          ByVal detailDictionary As Object, _
                                          ByVal taskCodeDictionary As Object, _
                                          ByVal taskKey As String, _
                                          ByVal intervalLabel As String, _
                                          ByVal detailText As String)

    If taskCodeDictionary.Exists(taskKey) Then Exit Sub

    AddSummaryItem summaryDictionary, detailDictionary, intervalLabel, detailText
    taskCodeDictionary.Add taskKey, True

End Sub
